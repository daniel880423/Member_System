            if Matrix[i][j] != 0:        #若右上方有不為0的就不是左下三角形

                left_up = False             #若不是則為False

                one＿dimensional＿matrix[index] = Matrix[length-1-i+j][i]      #算出位置後依序更改一維陣列中原本none的值
